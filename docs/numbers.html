    g = int(g * 255)
    b = int(b * 255)

    led.set_rgb_value(r, g, b)
    
    # Wait a bit
    time.sleep(duration_seconds_per_step)

    hue = hue + 1
    if hue > 360:
        hue = 1

led.set_rgb_value(0, 0, 0)
